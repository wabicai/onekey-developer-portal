export const TypeBadge = ({ type }) => {
  const styles = {
    major: 'bg-emerald-500/10 text-emerald-600 dark:bg-emerald-500/20 dark:text-emerald-400 border-emerald-500/20',
    minor: 'bg-blue-500/10 text-blue-600 dark:bg-blue-500/20 dark:text-blue-400 border-blue-500/20',
    patch: 'bg-zinc-500/10 text-zinc-600 dark:bg-zinc-500/20 dark:text-zinc-400 border-zinc-500/20'
  }
  const labels = { major: 'Major', minor: 'Feature', patch: 'Patch' }
  return (
    <span className={`inline-flex items-center px-2 py-0.5 text-[10px] font-semibold uppercase tracking-wider rounded border ${styles[type]}`}>
      {labels[type]}
    </span>
  )
}
